comparison$relationship <- ifelse(
  comparison$dmr_meandiff > 0 & comparison$atac_logFC < 0, "opposite_expected",
  ifelse(comparison$dmr_meandiff < 0 & comparison$atac_logFC > 0, "opposite_expected",
         "same_direction_or_other")
)
